DROP TABLE IF EXISTS OrderQty;
GO

CREATE TABLE OrderQty (Year SMALLINT, January INT, February INT, December INT);
GO

SELECT Year, Month, OrderQty
FROM (SELECT Year, January, February, December
	FROM dbo.OrderQty) qty
UNPIVOT (OrderQty FOR Month IN   
      (January, February, December)  
)AS unpvtOrderQty;  
GO
